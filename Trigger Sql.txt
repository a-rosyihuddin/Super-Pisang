Triger PAW
1. Update jumlahOrder
CREATE TRIGGER `update_jml_order` AFTER INSERT ON `order_details` FOR EACH ROW UPDATE orders SET total_order = (SELECT orders.total_order+new.jml_order FROM orders WHERE orders.id=new.order_id),
total_pembayaran = (SELECT orders.total_pembayaran+new.sub_total FROM orders WHERE orders.id=new.order_id)
WHERE orders.id=new.order_id
